    <script>
        const scrollContainer = document.querySelector('.scroll-container');
        const mainContent = document.getElementById('main-content');
        const galleryTrack = document.getElementById('gallery-track');
        const prevBtn = document.getElementById('prevBtn');
        const nextBtn = document.getElementById('nextBtn');
        let currentGalleryIndex = 0;

        if (scrollContainer) {
            window.onload = () => {
                scrollContainer.scrollLeft = mainContent.offsetLeft;
            };
        }

        if (galleryTrack) {
            const galleryItems = galleryTrack.querySelectorAll('.gallery-item');
            const totalItems = galleryItems.length;

            nextBtn.addEventListener('click', () => {
                currentGalleryIndex = (currentGalleryIndex + 1) % totalItems;
                updateGalleryPosition();
            });

            prevBtn.addEventListener('click', () => {
                currentGalleryIndex = (currentGalleryIndex - 1 + totalItems) % totalItems;
                updateGalleryPosition();
            });

            const updateGalleryPosition = () => {
                const itemWidth = galleryItems[0].offsetWidth;
                galleryTrack.style.transform = `translateX(${-currentGalleryIndex * itemWidth}px)`;
            };

            // Mobile gallery functionality
            const galleryTrackMobile = document.getElementById('gallery-track-mobile');
            const galleryItemsMobile = galleryTrackMobile.querySelectorAll('.gallery-item-mobile');
            const totalItemsMobile = galleryItemsMobile.length;
            let currentMobileIndex = 0;

            let startX = 0;
            let endX = 0;

            galleryTrackMobile.addEventListener('touchstart', (e) => {
                startX = e.touches[0].clientX;
            });

            galleryTrackMobile.addEventListener('touchmove', (e) => {
                endX = e.touches[0].clientX;
            });

            galleryTrackMobile.addEventListener('touchend', () => {
                const swipeThreshold = 50;
                if (startX - endX > swipeThreshold) {
                    currentMobileIndex = (currentMobileIndex + 1) % totalItemsMobile;
                } else if (endX - startX > swipeThreshold) {
                    currentMobileIndex = (currentMobileIndex - 1 + totalItemsMobile) % totalItemsMobile;
                }
                updateMobileGalleryPosition();
            });

            const updateMobileGalleryPosition = () => {
                const itemWidthMobile = galleryItemsMobile[0].offsetWidth;
                galleryTrackMobile.style.transform = `translateX(${-currentMobileIndex * itemWidthMobile}px)`;
            };
        }
        
        // Mobile navigation function
        window.scrollToSection = (sectionId) => {
            const section = document.getElementById(sectionId);
            if (section) {
                if (sectionId === 'galeria-mobile') {
                    // Specific scroll to mobile gallery
                    scrollContainer.scrollLeft = section.parentElement.parentElement.offsetLeft;
                    document.getElementById('gallery-container-mobile').scrollIntoView({ behavior: 'smooth' });
                } else {
                    scrollContainer.scrollLeft = section.offsetLeft;
                }
            }
        };
    </script>
