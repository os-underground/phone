import os
import re
from pathlib import Path

EXT_GROUPS = {
    'HTML': ['.html', '.htm'],
    'JSX': ['.jsx', '.tsx'],
    'JS': ['.js', '.mjs', '.cjs'],
    'CSS': ['.css'],
    'OTROS': []
}

def read_head(filepath, max_bytes=4000):
    try:
        with open(filepath, 'rb') as f:
            b = f.read(max_bytes)
        try:
            s = b.decode('utf-8', errors='ignore')
        except:
            s = b.decode('latin-1', errors='ignore')
        return s
    except Exception as e:
        return ''

def extract_title_from_text(text, ext):
    # html <title>
    if ext in ('.html', '.htm'):
        m = re.search(r'<title>(.*?)</title>', text, re.I|re.S)
        if m and m.group(1).strip():
            return m.group(1).strip()
        # h1
        m2 = re.search(r'<h1[^>]*>([^<]+)</h1>', text, re.I|re.S)
        if m2 and m2.group(1).strip():
            return m2.group(1).strip()
    # js/jsx comments: // Title: ...  or /* Title: ... */
    m3 = re.search(r'//\s*title[:\-]\s*(.+)', text, re.I)
    if m3 and m3.group(1).strip():
        return m3.group(1).strip()
    m4 = re.search(r'/\*\s*title[:\-]\s*(.+?)\s*\*/', text, re.I|re.S)
    if m4 and m4.group(1).strip():
        return m4.group(1).strip()
    # fallback: first non-empty line (short)
    for line in text.splitlines():
        line = line.strip()
        if not line: 
            continue
        if len(line) > 4 and len(line) < 120:
            # strip tags if any
            line = re.sub(r'<[^>]+>', '', line)
            return line.strip()[:120]
    return None

def group_for_ext(ext):
    for k, v in EXT_GROUPS.items():
        if ext in v:
            return k
    return 'OTROS'

def scan(path='.'):
    results = {k: [] for k in list(EXT_GROUPS.keys())+['OTROS']}
    for root, dirs, files in os.walk(path):
        # skip .git, node_modules, __pycache__
        if '.git' in root or 'node_modules' in root or '__pycache__' in root:
            continue
        for fname in files:
            # ignore this script and potential output
            if fname in ('generate_readme.py', 'README.md', 'index_preview.html', 'titulos.txt'):
                continue
            fpath = os.path.join(root, fname)
            ext = Path(fname).suffix.lower()
            group = group_for_ext(ext)
            text = read_head(fpath)
            title = extract_title_from_text(text, ext) or fname
            rel = os.path.relpath(fpath, path)
            results.setdefault(group, []).append({
                'file': rel.replace('\\','/'),
                'title': title
            })
    return results

📝 Encabezados # ## ###
En Markdown:

# al inicio de la línea = Título muy grande (nivel 1)

## = Título menos grande (nivel 2)

### = Subtítulo (nivel 3)

Tiene que ir un espacio después del #:

#Título (mal, sin espacio)
# Título (bien)
Si no dejas el espacio, GitHub lo muestra como texto normal y no como encabezado.

💻 Bloques de código con triple acento grave (```)
Los “comillas” que dices se llaman backticks o acentos graves.
Tres seguidos antes y después hacen un bloque de código:

<pre> ```html <h1>Hola mundo</h1> ``` </pre>
GitHub detecta “html” y colorea el código. Puedes usar bash, python, json, etc. para que coloree distinto.

✨ Estilo en Markdown
No se pueden usar <style> ni CSS directo en el README de GitHub Pages, porque GitHub limpia los estilos por seguridad.

Puedes, eso sí, usar HTML básico dentro del README para centrar, hacer tablas, imágenes, etc. (sin CSS avanzado).

Si necesitas CSS de verdad, debes crear un index.html con su <style> o linkar un .css.

🔍 Ver la vista previa
Cuando subas un README.md a GitHub, en la pestaña “Preview” del editor verás cómo quedará.
Para un chat tipo IA con Markdown, puedes usar librerías que rendericen Markdown (por ejemplo marked.js en React) y aplicarle tú tu propio CSS.
