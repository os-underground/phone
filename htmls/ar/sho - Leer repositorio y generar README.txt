def read_head(filepath, max_bytes=4000):
    try:
        with open(filepath, 'rb') as f:
            b = f.read(max_bytes)
        try:
            s = b.decode('utf-8', errors='ignore')
        except:
            s = b.decode('latin-1', errors='ignore')
        return s
    except Exception as e:
        return ''

def extract_title_from_text(text, ext):
    # html <title>
    if ext in ('.html', '.htm'):
        m = re.search(r'<title>(.*?)</title>', text, re.I|re.S)
        if m and m.group(1).strip():
            return m.group(1).strip()
        # h1
        m2 = re.search(r'<h1[^>]*>([^<]+)</h1>', text, re.I|re.S)
        if m2 and m2.group(1).strip():
            return m2.group(1).strip()
    # js/jsx comments: // Title: ...  or /* Title: ... */
    m3 = re.search(r'//\s*title[:\-]\s*(.+)', text, re.I)
    if m3 and m3.group(1).strip():
        return m3.group(1).strip()
    m4 = re.search(r'/\*\s*title[:\-]\s*(.+?)\s*\*/', text, re.I|re.S)
    if m4 and m4.group(1).strip():
        return m4.group(1).strip()
    # fallback: first non-empty line (short)
    for line in text.splitlines():
        line = line.strip()
        if not line: 
            continue
        if len(line) > 4 and len(line) < 120:
            # strip tags if any
            line = re.sub(r'<[^>]+>', '', line)
            return line.strip()[:120]
    return None

def group_for_ext(ext):
    for k, v in EXT_GROUPS.items():
        if ext in v:
            return k
    return 'OTROS'

def scan(path='.'):
    results = {k: [] for k in list(EXT_GROUPS.keys())+['OTROS']}
    for root, dirs, files in os.walk(path):
        # skip .git, node_modules, __pycache__
        if '.git' in root or 'node_modules' in root or '__pycache__' in root:
            continue
        for fname in files:
            # ignore this script and potential output
            if fname in ('generate_readme.py', 'README.md', 'index_preview.html', 'titulos.txt'):
                continue
            fpath = os.path.join(root, fname)
            ext = Path(fname).suffix.lower()
            group = group_for_ext(ext)
            text = read_head(fpath)
            title = extract_title_from_text(text, ext) or fname
            rel = os.path.relpath(fpath, path)
            results.setdefault(group, []).append({
                'file': rel.replace('\\','/'),
                'title': title
            })
    return results
